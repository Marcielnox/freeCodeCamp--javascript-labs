function pyramid(character, count, inverted) {
  let rows = [];

  for (let i = 1; i <= count; i++) {
    let spaces = " ".repeat(count - i);
    let chars = character.repeat(2 * i - 1);
    let row = spaces + chars;
    if (inverted) {
      rows.unshift(row);
    } else {
      rows.push(row);
    }
  }
  return "\n" + rows.join("\n") + "\n";
}
