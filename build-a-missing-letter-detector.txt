function fearNotLetter(str) {
  for (let i = 0; i < str.length - 1; i++) {
    let code = str.charCodeAt(i);
    let nextCode = str.charCodeAt(i + 1);
    if (nextCode !== code + 1) {
      return String.fromCharCode(code + 1);
    }
  }
  return undefined;
}
