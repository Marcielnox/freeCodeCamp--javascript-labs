function largestOfAll(arr) {
  let results = [];

  for (let i = 0; i < arr.length; i++) {
    let largestNum = arr[i][0];

    for (let j = 1; j < arr[i].length; j++) {
      if (arr[i][j] > largestNum) {
        largestNum = arr[i][j];
      }
    }
    results.push(largestNum);
  }

  return results;
}
