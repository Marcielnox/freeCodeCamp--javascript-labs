** start of script.js **

function validateForm() {

    const fullName = document.getElementById('full-name');
    const email = document.getElementById('email');
    const orderNo = document.getElementById('order-no');
    const productCode = document.getElementById('product-code');
    const quantity = document.getElementById('quantity');

    const complaintGroup = document.getElementById('complaints-group');
    const otherComplaint = document.getElementById('other-complaint');
    const complaintDesc = document.getElementById('complaint-description');

    const solutionGroup = document.getElementById('solutions-group');
    const otherSolution = document.getElementById('other-solution');
    const solutionDesc = document.getElementById('solution-description');

    const results = {
        "full-name": fullName.value.trim() !== "",
        "email": /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email.value),
        "order-no": /^2024\d{6}$/.test(orderNo.value),
        "product-code": /^[a-zA-Z]{2}\d{2}-[a-zA-Z]\d{3}-[a-zA-Z]{2}\d$/.test(productCode.value),
        "quantity": Number.isInteger(Number(quantity.value)) && parseInt(quantity.value) > 0,
        "complaints-group": !!complaintGroup.querySelector('input:checked'),
        "complaint-description": otherComplaint.checked ? complaintDesc.value.trim().length >= 20 : true,
        "solutions-group": !!solutionGroup.querySelector('input:checked'),
        "solution-description": otherSolution.checked ? solutionDesc.value.trim().length >= 20 : true
    };

    return results;
}

function isValid(validationObject) {
    return Object.values(validationObject).every(val => val === true);
}

const inputs = ['full-name', 'email', 'order-no', 'product-code', 'quantity', 'complaint-description', 'solution-description'];
const groups = ['complaints-group', 'solutions-group'];

inputs.forEach(id => {
    document.getElementById(id).addEventListener('change', () => {
        const results = validateForm();
        const element = document.getElementById(id);
        element.style.borderColor = results[id] ? "green" : "red";
    });
});

groups.forEach(id => {
    document.getElementById(id).addEventListener('change', () => {
        const results = validateForm();
        const element = document.getElementById(id);
        element.style.borderColor = results[id] ? "green" : "red";
    });
});

document.querySelector('form').addEventListener('submit', (e) => {
    const results = validateForm();
    if (!isValid(results)) {
        e.preventDefault();
        Object.keys(results).forEach(key => {
            const el = document.getElementById(key);
            if (!results[key]) el.style.borderColor = "red";
        });
    }
});
