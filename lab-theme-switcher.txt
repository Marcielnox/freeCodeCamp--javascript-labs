** start of script.js **

const themesSwitcher = document.getElementById('theme-switcher-button');
const themeDropDown = document.getElementById('theme-dropdown');
const themeBlue = document.getElementById('theme-blue');
const themeRed = document.getElementById('theme-red');

const themes = [
  {name: "blue", message: "Blue Theme"},
  {name: "red", message: "Red Theme"}
];

themesSwitcher.addEventListener("click", () => {
  themeDropDown.toggleAttribute("hidden");
  const isClosed = themeDropDown.hasAttribute("hidden");
  themesSwitcher.setAttribute("aria-expanded", isClosed ? "false" : "true");
});
themeBlue.addEventListener("click", () => {
  document.body.classList.remove("theme-red");
  document.body.classList.add("theme-blue");
});
themeRed.addEventListener("click", () => {
  document.body.classList.remove("theme-blue");
  document.body.classList.add("theme-red");
});
const themeMsg = document.getElementById('aria-message');
themeBlue.addEventListener("click", () => {
  const blueTheme = themes.find(t => t.name === "blue");
  
  document.body.className = `theme-${blueTheme.name}`;
  themeMsg.textContent = blueTheme.message;
});
themeRed.addEventListener("click", () => {
  const redTheme = themes.find(t => t.name === "red");
  
  document.body.className = `theme-${redTheme.name}`;
  themeMsg.textContent = redTheme.message;
});

** end of script.js **
